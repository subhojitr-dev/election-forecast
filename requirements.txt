# Election Forecast Dashboard — Python dependencies
# Install:  pip install -r requirements.txt

pandas        # ETL (Phase 1) — read CSVs, aggregate, load SQLite
clarify       # Phase 2 — parse Clarity ENR XML feeds (pulls in lxml, requests)
requests      # Phase 2 — HTTP client for live feed fetching

fastapi       # Phase 4 — REST API + WebSocket
uvicorn       # Phase 4 — ASGI server

httpx         # Live-readiness — state live feeds (nc_*, ga_, pa_, az_live_feed): plain HTTP + JSON/zip
playwright    # mi_live_feed.py — MI's Cloudflare blocks headless + plain httpx; needs a
              # HEADED browser (see ingestor/mi_live_feed.py docstring). Local/election-night
              # only — NOT needed on the Render server (MI ingest runs from a machine with a
              # display, not the API host). Run `playwright install chromium` after pip install.
